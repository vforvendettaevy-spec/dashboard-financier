function formatEuro(val, signe = false) {
  if (!val || isNaN(val)) return '0,00 €';
  const fmt = val.toLocaleString('fr-FR', { minimumFractionDigits: 2, maximumFractionDigits: 2 }) + ' €';
  return signe && val >= 0 ? '+ ' + fmt : fmt;
}

function formatEuroRound(val) {
  if (!val || isNaN(val)) return '0 €';
  return Math.round(val).toLocaleString('fr-FR') + ' €';
}

function setPV(id, val) {
  const el = $(id);
  if (!el) return;
  el.textContent = formatEuro(val, true);
  el.style.color = val >= 0 ? 'var(--green)' : 'var(--red)';
}

function setPerf(id, val) {
  const el = $(id);
  if (!el) return;
  el.textContent = (val >= 0 ? '+' : '') + val.toFixed(2) + '%';
  el.style.color = val >= 0 ? 'var(--green)' : 'var(--red)';
}

function projeterCapital(capitalInitial, invMensuel, mois, rendAnnuel) {
  const mRate = (rendAnnuel / 100) / 12;
  if (mRate === 0) return capitalInitial + invMensuel * mois;
  return capitalInitial * Math.pow(1 + mRate, mois) + invMensuel * ((Math.pow(1 + mRate, mois) - 1) / mRate);
}

function calculerValeurPEA(rendAnnuel) {
  const today = new Date();
  const rend = rendAnnuel / 100;
  const fraisPct = $val('fraisCourtage') / 100;
  let valeur = 0, totalVerse = 0, totalFrais = 0;
  state.historiquePEA.forEach(item => {
    const frais = item.montant * fraisPct;
    totalFrais += frais;
    totalVerse += item.montant;
    if (item.partsWPEA && item.partsPAEEM && state.coursWPEA > 0 && state.coursPAEEM > 0) {
      valeur += item.partsWPEA * state.coursWPEA + item.partsPAEEM * state.coursPAEEM;
    } else {
      const depDate = new Date(item.date);
      const months = Math.max(0, (today.getFullYear() - depDate.getFullYear()) * 12 + (today.getMonth() - depDate.getMonth()));
      valeur += (item.montant - frais) * Math.pow(1 + rend / 12, months);
    }
  });
  return { valeur, totalVerse, plusValue: valeur - (totalVerse - totalFrais), totalFrais };
}

function calculerValeurCTO(rendAnnuel) {
  const today = new Date();
  const rend = rendAnnuel / 100;
  const fraisPct = $val('fraisCourtage') / 100;
  let valeur = 0, totalVerse = 0, totalFrais = 0;
  state.historiqueCTO.forEach(item => {
    const frais = item.montant * fraisPct;
    totalFrais += frais;
    totalVerse += item.montant;
    if (item.partsWPEA && item.partsPAEEM && state.coursEUNL > 0 && state.coursIS3N > 0) {
      valeur += item.partsWPEA * state.coursEUNL + item.partsPAEEM * state.coursIS3N;
    } else {
      const depDate = new Date(item.date);
      const months = Math.max(0, (today.getFullYear() - depDate.getFullYear()) * 12 + (today.getMonth() - depDate.getMonth()));
      valeur += (item.montant - frais) * Math.pow(1 + rend / 12, months);
    }
  });
  return { valeur, totalVerse, plusValue: valeur - (totalVerse - totalFrais), totalFrais };
}

function calculerCrypto(historique, coursActuel) {
  const fraisPct = $val('fraisCourtage') / 100;
  let totalInvesti = 0, quantiteTotale = 0, totalFrais = 0;
  historique.forEach(item => {
    const frais = item.montant * fraisPct;
    totalFrais += frais;
    quantiteTotale += (item.montant - frais) / item.prixAchat;
    totalInvesti += item.montant;
  });
  const valeurActuelle = quantiteTotale * coursActuel;
  return { totalInvesti, quantiteTotale, valeurActuelle, plusValue: valeurActuelle - (totalInvesti - totalFrais), totalFrais };
}

    totalCrypto: btc.valeurActuelle + eth.valeurActuelle,

  };
}

    if (!response.ok) throw new Error('HTTP ' + response.status);
    const data = await response.json();
    if (!data.bitcoin || !data.ethereum) throw new Error('Données incomplètes');

    state.coursBTC = data.bitcoin.eur;
    state.coursETH = data.ethereum.eur;
    $('prixBTC').textContent = 'Prix: ' + formatEuro(state.coursBTC);
    $('prixETH').textContent = 'Prix: ' + formatEuro(state.coursETH);
    $('nouveauPrixBTC').value = state.coursBTC.toFixed(2);
    $('nouveauPrixETH').value = state.coursETH.toFixed(2);
    $('lastUpdate').textContent = new Date().toLocaleString('fr-FR');
    calculer();
  } catch(e) {
    $('prixBTC').textContent = 'Prix: Entre manuellement';
    $('prixETH').textContent = 'Prix: Entre manuellement';
    $('lastUpdate').textContent = '⚠️ API indisponible';
  }

  if (btn) { btn.textContent = '🔄 Rafraîchir les cours'; btn.disabled = false; }
}

// PIE CHART
// =============================================
function dessinerPieChart(valPEA, valCTO, valCrypto, livret, courant, especes) {
  const canvas = $('pieChart'), ctx = canvas.getContext('2d');
  const total = valPEA + valCTO + valCrypto + livret + courant + especes;
  const legend = $('pieLegend');
  if (total === 0) {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#6b7280'; ctx.font = '16px DM Sans,sans-serif'; ctx.textAlign = 'center';
    ctx.fillText('Commence à investir !', canvas.width / 2, canvas.height / 2);
    legend.innerHTML = ''; return;
  }
  const data = [
    { label: '📈 PEA', value: valPEA, color: '#7c6dfa' },
    { label: '💵 PEA Espèces', value: especes, color: '#a78bfa' },
    { label: '📊 CTO', value: valCTO, color: '#6366f1' },
    { label: '₿ Crypto', value: valCrypto, color: '#fb923c' },
    { label: '💚 Livret A', value: livret, color: '#00d4aa' },
    { label: '💳 Compte', value: courant, color: '#60a5fa' }
  ].filter(x => x.value > 0);

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  const cx = canvas.width / 2, cy = canvas.height / 2, r = 110;
  let angle = -Math.PI / 2;
  data.forEach(item => {
    const slice = (item.value / total) * 2 * Math.PI;
    ctx.beginPath(); ctx.fillStyle = item.color; ctx.moveTo(cx, cy);
    ctx.arc(cx, cy, r, angle, angle + slice); ctx.closePath(); ctx.fill();
    const mid = angle + slice / 2;
    ctx.fillStyle = '#fff'; ctx.font = 'bold 12px Outfit,sans-serif'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
    ctx.fillText(((item.value / total) * 100).toFixed(1) + '%', cx + Math.cos(mid) * r * 0.65, cy + Math.sin(mid) * r * 0.65);
    angle += slice;
  });
  legend.innerHTML = data.map(item => {
    const pct = ((item.value / total) * 100).toFixed(1);
    return `<div class="legend-item"><div class="legend-color" style="background:${item.color}"></div><span class="legend-label">${item.label}</span><span class="legend-value">${formatEuro(item.value)} (${pct}%)</span></div>`;
  }).join('');
}

// HISTORIQUES
// =============================================
function afficherHistorique(type) {
  const isPEA = type === 'pea';
  const hist = isPEA ? state.historiquePEA : state.historiqueCTO;
  const tbody = $(isPEA ? 'historiquePEA' : 'historiqueCTO');
  if (hist.length === 0) { tbody.innerHTML = '<tr><td colspan="10" style="text-align:center;color:var(--text-muted);padding:24px">Aucun investissement</td></tr>'; return; }

  const rend = $val('rendement') / 100, fraisPct = $val('fraisCourtage') / 100, today = new Date();
  const coursW = isPEA ? state.coursWPEA : state.coursEUNL;
  const coursE = isPEA ? state.coursPAEEM : state.coursIS3N;
  let cumulValeur = 0, cumulVerse = 0;

  tbody.innerHTML = hist.map(item => {
    const frais = item.montant * fraisPct;
    let valeurLigne = 0;
    if (item.partsWPEA && item.partsPAEEM && coursW > 0 && coursE > 0) {
      valeurLigne = item.partsWPEA * coursW + item.partsPAEEM * coursE;
    } else {
      const months = Math.max(0, (today.getFullYear() - new Date(item.date).getFullYear()) * 12 + (today.getMonth() - new Date(item.date).getMonth()));
      valeurLigne = (item.montant - frais) * Math.pow(1 + rend / 12, months);
    }
    cumulValeur += valeurLigne; cumulVerse += item.montant;
    const pvCumul = cumulValeur - (cumulVerse - cumulVerse * fraisPct);
    const dateStr = new Date(item.date).toLocaleDateString('fr-FR', { day: '2-digit', month: 'long', year: 'numeric' });
    const pw = item.partsWPEA > 0 ? item.partsWPEA.toFixed(4) : '--';
    const ppw = item.prixAchatWPEA > 0 ? formatEuro(item.prixAchatWPEA) : '--';
    const pe = item.partsPAEEM > 0 ? item.partsPAEEM.toFixed(4) : '--';
    const ppe = item.prixAchatPAEEM > 0 ? formatEuro(item.prixAchatPAEEM) : '--';
    return `<tr><td>${dateStr}</td><td><strong>${formatEuro(item.montant)}</strong></td><td style="color:var(--red)">${formatEuro(frais)}</td><td>${pw}</td><td>${ppw}</td><td>${pe}</td><td>${ppe}</td><td class="positive"><strong>${formatEuro(cumulValeur)}</strong></td><td class="${pvCumul >= 0 ? 'positive' : 'negative'}">${formatEuro(pvCumul, true)}</td><td><button class="btn-delete" onclick="supprimer('${type}',${item.id})">🗑️</button></td></tr>`;
  }).join('');
}

function afficherHistoriqueCrypto(type) {
  const h = type === 'btc' ? state.historiqueBTC : state.historiqueETH;
  const tbody = $('historique' + (type === 'btc' ? 'BTC' : 'ETH'));
  const cours = type === 'btc' ? state.coursBTC : state.coursETH;
  const fraisPct = $val('fraisCourtage') / 100;
  if (h.length === 0) { tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:var(--text-muted);padding:24px">Aucun investissement</td></tr>'; return; }
  tbody.innerHTML = h.map(item => {
    const frais = item.montant * fraisPct, montantNet = item.montant - frais;
    const quantite = montantNet / item.prixAchat, valeurActuelle = quantite * cours, plusValue = valeurActuelle - montantNet;
    const dateStr = new Date(item.date).toLocaleDateString('fr-FR', { day: '2-digit', month: 'long', year: 'numeric' });
    return `<tr><td>${dateStr}</td><td><strong>${formatEuro(item.montant)}</strong></td><td>${formatEuro(item.prixAchat)}</td><td>${quantite.toFixed(8)}</td><td class="positive">${formatEuro(valeurActuelle)}</td><td class="${plusValue >= 0 ? 'positive' : 'negative'}">${formatEuro(plusValue, true)}</td><td><button class="btn-delete" onclick="supprimer('${type}',${item.id})">🗑️</button></td></tr>`;
  }).join('');
}

function afficherProjection(capital, rend) {
  const ages = [20, 25, 30, 35, 40, 45, 50, 55, 60];
  const age = $val('age'), invPEA = $val('investissementPEA'), invCTO = $val('investissementCTO');
  const fraisPct = $val('fraisCourtage') / 100;
  const inv = (invPEA + invCTO) * (1 - fraisPct);
  const grid = $('projectionGrid'), tbody = $('projectionBody');
  grid.innerHTML = ''; tbody.innerHTML = '';

  ages.forEach(a => {
    if (a <= age) return;
    const yrs = a - age, mths = yrs * 12;
    const final = projeterCapital(capital, inv, mths, rend);
    const versed = capital + (invPEA + invCTO) * mths;
    const pv = final - versed + (invPEA + invCTO) * mths * fraisPct;
    grid.innerHTML += `<div class="projection-card"><div class="projection-age">${a} ans</div><div class="projection-amount">${formatEuro(final)}</div></div>`;
    tbody.innerHTML += `<tr><td><strong>${a} ans</strong></td><td>${yrs} an${yrs > 1 ? 's' : ''}</td><td>${formatEuro(versed)}</td><td class="positive"><strong>${formatEuro(final)}</strong></td><td class="${pv >= 0 ? 'positive' : 'negative'}">${formatEuro(pv, true)}</td></tr>`;
  });
}

function calculerScenarios(capital) {
  const age = $val('age'), invPEA = $val('investissementPEA'), invCTO = $val('investissementCTO');
  const fraisPct = $val('fraisCourtage') / 100;
  const inv = (invPEA + invCTO) * (1 - fraisPct);
  const yrs = state.ageObjectif - age;
  if (yrs <= 0) return;
  const mths = yrs * 12;

  [{ id: 'Pessimiste', r: 5 }, { id: 'Neutre', r: 6.5 }, { id: 'Optimiste', r: 8 }, { id: 'UltraOptimiste', r: 9.5 }].forEach(s => {
    const final = projeterCapital(capital, inv, mths, s.r);
    const gains = final - capital - (invPEA + invCTO) * mths + (invPEA + invCTO) * mths * fraisPct;
    $('scenario' + s.id).textContent = formatEuro(final);
    $('detail' + s.id).textContent = '+' + formatEuro(gains) + ' de gains';
  });
}

// GRAPHIQUE EVOLUTION (AMELIORÉ)
// =============================================
function dessinerGraphiqueEvolution() {
  const canvas = $('evolutionChart');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  const age = $val('age'), rend = $val('rendement');
  const invPEA = $val('investissementPEA'), invCTO = $val('investissementCTO');
  const fraisPct = $val('fraisCourtage') / 100;
  const data = getCapitalTotal();
  const capitalActuel = data.pea.valeur + data.cto.valeur + data.livret + data.courant + data.especes;
  const objectif = state.objectifCapital;

  // Générer points chaque année
  const points = [];
  for (let a = age; a <= 65; a++) {
    const yrs = a - age, mths = yrs * 12;
    const inv = (invPEA + invCTO) * (1 - fraisPct);
    const capital = projeterCapital(capitalActuel, inv, mths, rend);
    const verse = capitalActuel + (invPEA + invCTO) * mths;
    points.push({ age: a, capital, verse });
  }

  const padding = { top: 40, right: 30, bottom: 50, left: 80 };
  const gw = canvas.width - padding.left - padding.right;
  const gh = canvas.height - padding.top - padding.bottom;
  const maxVal = Math.max(...points.map(p => p.capital), objectif * 1.05, 10000);

  // Grille
  ctx.strokeStyle = 'rgba(255,255,255,0.04)';
  ctx.lineWidth = 1;
  for (let i = 0; i <= 6; i++) {
    const y = padding.top + (gh / 6) * i;
    ctx.beginPath(); ctx.moveTo(padding.left, y); ctx.lineTo(canvas.width - padding.right, y); ctx.stroke();
    // Labels Y
    const val = maxVal - (maxVal / 6) * i;
    ctx.fillStyle = 'rgba(200,196,255,0.4)'; ctx.font = '10px Outfit,sans-serif';
    ctx.textAlign = 'right'; ctx.textBaseline = 'middle';
    ctx.fillText(val >= 1000000 ? (val / 1000000).toFixed(1) + 'M €' : (val / 1000).toFixed(0) + 'k €', padding.left - 10, y);
  }

  const getX = i => padding.left + (gw / (points.length - 1)) * i;
  const getY = val => padding.top + gh - (val / maxVal) * gh;

  // Ligne objectif (pointillée orange)
  if (objectif <= maxVal) {
    const objY = getY(objectif);
    ctx.save(); ctx.setLineDash([6, 4]); ctx.strokeStyle = 'rgba(255,140,66,0.4)'; ctx.lineWidth = 1.5;
    ctx.beginPath(); ctx.moveTo(padding.left, objY); ctx.lineTo(canvas.width - padding.right, objY); ctx.stroke(); ctx.restore();
    ctx.fillStyle = 'rgba(255,140,66,0.7)'; ctx.font = 'bold 10px Outfit,sans-serif'; ctx.textAlign = 'right'; ctx.textBaseline = 'bottom';
    ctx.fillText('🎯 Objectif ' + formatEuroRound(objectif), canvas.width - padding.right, objY - 5);
  }

  // Aire "total versé" (vert foncé)
  ctx.beginPath(); ctx.moveTo(getX(0), getY(0));
  points.forEach((p, i) => ctx.lineTo(getX(i), getY(p.verse)));
  ctx.lineTo(getX(points.length - 1), getY(0)); ctx.closePath();
  const gradVerse = ctx.createLinearGradient(0, padding.top, 0, canvas.height - padding.bottom);
  gradVerse.addColorStop(0, 'rgba(0,210,127,0.15)'); gradVerse.addColorStop(1, 'rgba(0,210,127,0)');
  ctx.fillStyle = gradVerse; ctx.fill();

  // Aire "capital" (violet)
  ctx.beginPath(); ctx.moveTo(getX(0), getY(0));
  points.forEach((p, i) => ctx.lineTo(getX(i), getY(p.capital)));
  ctx.lineTo(getX(points.length - 1), getY(0)); ctx.closePath();
  const gradCap = ctx.createLinearGradient(0, padding.top, 0, canvas.height - padding.bottom);
  gradCap.addColorStop(0, 'rgba(139,119,255,0.2)'); gradCap.addColorStop(1, 'rgba(139,119,255,0)');
  ctx.fillStyle = gradCap; ctx.fill();

  // Ligne "total versé"
  ctx.beginPath(); ctx.strokeStyle = '#00D27F'; ctx.lineWidth = 2; ctx.lineJoin = 'round';
  points.forEach((p, i) => { if (i === 0) ctx.moveTo(getX(i), getY(p.verse)); else ctx.lineTo(getX(i), getY(p.verse)); });
  ctx.stroke();

  // Ligne "capital"
  ctx.beginPath(); ctx.strokeStyle = '#8B77FF'; ctx.lineWidth = 3; ctx.lineJoin = 'round';
  points.forEach((p, i) => { if (i === 0) ctx.moveTo(getX(i), getY(p.capital)); else ctx.lineTo(getX(i), getY(p.capital)); });
  ctx.stroke();

  // Points + labels tous les 5 ans
  points.forEach((p, i) => {
    if (p.age % 5 !== 0 && i !== 0 && i !== points.length - 1) return;
    const x = getX(i), y = getY(p.capital);
    // Point
    ctx.beginPath(); ctx.fillStyle = '#8B77FF'; ctx.arc(x, y, 5, 0, 2 * Math.PI); ctx.fill();
    ctx.beginPath(); ctx.fillStyle = '#fff'; ctx.arc(x, y, 2, 0, 2 * Math.PI); ctx.fill();
    // Label
    ctx.fillStyle = 'rgba(240,240,255,0.85)'; ctx.font = 'bold 10px Outfit,sans-serif'; ctx.textAlign = 'center'; ctx.textBaseline = 'bottom';
    const label = p.capital >= 1000000 ? (p.capital / 1000000).toFixed(1) + 'M' : (p.capital / 1000).toFixed(0) + 'k';
    ctx.fillText(label, x, y - 10);
    // Axe X
    ctx.fillStyle = 'rgba(200,196,255,0.5)'; ctx.font = '10px Outfit,sans-serif'; ctx.textBaseline = 'top';
    ctx.fillText(p.age + ' ans', x, canvas.height - padding.bottom + 10);
  });
}

function calculerDCA() {
  const calcDCA = (hist, keyParts, keyPrix) => {
    let totalParts = 0, totalInvesti = 0;
    hist.forEach(item => { if (item[keyParts] > 0) { totalParts += item[keyParts]; totalInvesti += item[keyParts] * item[keyPrix]; } });
    return totalParts > 0 ? totalInvesti / totalParts : 0;
  };
  const fraisPct = $val('fraisCourtage') / 100;
  const calcDCACrypto = (hist) => {
    let totalInv = 0, totalQte = 0;
    hist.forEach(item => { const n = item.montant - item.montant * fraisPct; totalQte += n / item.prixAchat; totalInv += n; });
    return totalQte > 0 ? totalInv / totalQte : 0;
  };

  const dcaW = calcDCA(state.historiquePEA, 'partsWPEA', 'prixAchatWPEA');
  const dcaP = calcDCA(state.historiquePEA, 'partsPAEEM', 'prixAchatPAEEM');
  const dcaE = calcDCA(state.historiqueCTO, 'partsWPEA', 'prixAchatWPEA');
  const dcaI = calcDCA(state.historiqueCTO, 'partsPAEEM', 'prixAchatPAEEM');
  $('dcaWPEA').textContent = dcaW > 0 ? formatEuro(dcaW) : '-- €';
  $('dcaPAEEM').textContent = dcaP > 0 ? formatEuro(dcaP) : '-- €';
  $('dcaWPEACTO').textContent = dcaE > 0 ? formatEuro(dcaE) : '-- €';
  $('dcaPAEEMCTO').textContent = dcaI > 0 ? formatEuro(dcaI) : '-- €';
  $('dcaBTC').textContent = state.historiqueBTC.length > 0 ? formatEuro(calcDCACrypto(state.historiqueBTC)) : '-- €';
  $('dcaETH').textContent = state.historiqueETH.length > 0 ? formatEuro(calcDCACrypto(state.historiqueETH)) : '-- €';
}

function calculerSimulation() {
  const invSupp = $val('simInvSupp'), rendSim = $val('simRendement'), ageCible = $val('simAgeCible'), age = $val('age');
  const invPEA = $val('investissementPEA'), invCTO = $val('investissementCTO'), fraisPct = $val('fraisCourtage') / 100;
  const data = getCapitalTotal();
  const capitalActuel = data.pea.valeur + data.cto.valeur + data.livret + data.courant + data.especes;
  const yrs = ageCible - age;
  if (yrs <= 0) { alert('⚠️ Âge cible doit être supérieur à ton âge !'); return; }
  const mths = yrs * 12;
  const capActuel = projeterCapital(capitalActuel, (invPEA + invCTO) * (1 - fraisPct), mths, rendSim);
  const capSimule = projeterCapital(capitalActuel, (invPEA + invCTO + invSupp) * (1 - fraisPct), mths, rendSim);
  $('resultatsSimulation').innerHTML = `
    <div class="stat-card"><div class="stat-label">💰 Capital actuel</div><div class="stat-value">${formatEuro(capitalActuel)}</div></div>
    <div class="stat-card"><div class="stat-label">📊 Scénario actuel à ${ageCible} ans</div><div class="stat-value">${formatEuro(capActuel)}</div></div>
    <div class="stat-card"><div class="stat-label">🚀 Avec +${formatEuroRound(invSupp)}/mois</div><div class="stat-value">${formatEuro(capSimule)}</div></div>
    <div class="stat-card"><div class="stat-label">💎 Différence</div><div class="stat-value positive">+ ${formatEuro(capSimule - capActuel)}</div></div>`;
}

function calculerFIRE() {
  const depenses = $val('fireDepenses'), taux = $val('fireTaux'), age = $val('age');
  const invPEA = $val('investissementPEA'), invCTO = $val('investissementCTO'), rend = $val('rendement'), fraisPct = $val('fraisCourtage') / 100;
  const data = getCapitalTotal();
  const capitalActuel = data.pea.valeur + data.cto.valeur + data.livret + data.courant + data.especes;
  const capitalNecessaire = depenses / (taux / 100);
  let ageFIRE = age;
  const manquant = capitalNecessaire - capitalActuel;
  if (manquant > 0) {
    const invMensuel = (invPEA + invCTO) * (1 - fraisPct), mRate = (rend / 100) / 12;
    if (invMensuel > 0 && mRate > 0) {
      const moisN = Math.log((capitalNecessaire * mRate / invMensuel) + 1) / Math.log(1 + mRate) - Math.log((capitalActuel * mRate / invMensuel) + 1) / Math.log(1 + mRate);
      ageFIRE = age + Math.ceil(moisN / 12);
    } else ageFIRE = '∞';
  }
  $('resultatsFIRE').innerHTML = `
    <div class="stat-card"><div class="stat-label">🎯 Capital FIRE nécessaire</div><div class="stat-value">${formatEuro(capitalNecessaire)}</div></div>
    <div class="stat-card"><div class="stat-label">💰 Capital actuel</div><div class="stat-value">${formatEuro(capitalActuel)}</div></div>
    <div class="stat-card"><div class="stat-label">${manquant > 0 ? '📉 Manquant' : '✅ Surplus'}</div><div class="stat-value" style="color:${manquant > 0 ? 'var(--orange)' : 'var(--green)'}">${formatEuro(Math.abs(manquant))}</div></div>
    <div class="stat-card"><div class="stat-label">🔥 Âge FIRE estimé</div><div class="stat-value" style="color:var(--green)">${ageFIRE === '∞' ? '∞' : ageFIRE + ' ans'}</div></div>`;
}

function calculerPerformance() {
  const rend = $val('rendement');
  const pea = calculerValeurPEA(rend), cto = calculerValeurCTO(rend);
  const btc = calculerCrypto(state.historiqueBTC, state.coursBTC), eth = calculerCrypto(state.historiqueETH, state.coursETH);
  const totalInvesti = pea.totalVerse + cto.totalVerse + btc.totalInvesti + eth.totalInvesti;
  const totalValeur = pea.valeur + cto.valeur + btc.valeurActuelle + eth.valeurActuelle;
  const perfTotale = totalInvesti > 0 ? ((totalValeur - totalInvesti) / totalInvesti) * 100 : 0;
  setPerf('perfTotal', perfTotale);
  setPerf('perfPEA', pea.totalVerse > 0 ? ((pea.valeur - pea.totalVerse) / pea.totalVerse) * 100 : 0);
  setPerf('perfCTO', cto.totalVerse > 0 ? ((cto.valeur - cto.totalVerse) / cto.totalVerse) * 100 : 0);
  setPerf('perfCrypto', (btc.totalInvesti + eth.totalInvesti) > 0 ? (((btc.valeurActuelle + eth.valeurActuelle) - (btc.totalInvesti + eth.totalInvesti)) / (btc.totalInvesti + eth.totalInvesti)) * 100 : 0);

  const allTx = [...state.historiquePEA, ...state.historiqueCTO, ...state.historiqueBTC, ...state.historiqueETH];
  let anciennete = 0;
  if (allTx.length > 0) anciennete = (new Date() - new Date(Math.min(...allTx.map(t => new Date(t.date))))) / (1000 * 60 * 60 * 24 * 365);
  const cagr = anciennete > 0 ? (Math.pow(totalValeur / Math.max(totalInvesti, 1), 1 / anciennete) - 1) * 100 : 0;
  setPerf('perfCAGR', cagr);

  const diff = cagr - 7;
  const vb = $('vsBenchmark');
  vb.textContent = (diff >= 0 ? '+' : '') + diff.toFixed(2) + '%';
  vb.style.color = diff >= 0 ? 'var(--green)' : 'var(--red)';

  // Exposition actifs
  const totalAll = pea.valeur + cto.valeur + btc.valeurActuelle + eth.valeurActuelle;
  const expositions = [];
  if (pea.valeur > 0) { expositions.push({ label: '📈 PEA WPEA', value: pea.valeur * 0.85, pct: totalAll > 0 ? (pea.valeur * 0.85 / totalAll) * 100 : 0 }); expositions.push({ label: '📈 PEA PAEEM', value: pea.valeur * 0.15, pct: totalAll > 0 ? (pea.valeur * 0.15 / totalAll) * 100 : 0 }); }

(cto.valeur * pw / totalAll) * 100 : 0 }); expositions.push({ label: '📊 CTO IS3N', value: cto.valeur * pe, pct: totalAll > 0 ? (cto.valeur * pe / totalAll) * 100 : 0 }); }
  if (btc.valeurActuelle > 0) expositions.push({ label: '₿ Bitcoin', value: btc.valeurActuelle, pct: totalAll > 0 ? (btc.valeurActuelle / totalAll) * 100 : 0 });
  if (eth.valeurActuelle > 0) expositions.push({ label: '⟠ Ethereum', value: eth.valeurActuelle, pct: totalAll > 0 ? (eth.valeurActuelle / totalAll) * 100 : 0 });
  expositions.sort((a, b) => b.value - a.value);
  const htmlExp = $('expositionActifs');

function dessinerExpositionGeo() {
  const canvas = $('geoChart');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const rend = $val('rendement');
  const pea = calculerValeurPEA(rend), cto = calculerValeurCTO(rend);
  const btc = calculerCrypto(state.historiqueBTC, state.coursBTC), eth = calculerCrypto(state.historiqueETH, state.coursETH);
  const total = pea.valeur + cto.valeur + btc.valeurActuelle + eth.valeurActuelle;
  if (total === 0) { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = 'var(--text-muted)'; ctx.font = '14px DM Sans'; ctx.textAlign = 'center'; ctx.fillText('Aucune donnée', canvas.width / 2, canvas.height / 2); return; }
  const pw = ($val('ctoMsciWorld') || 90) / 100, pe = ($val('ctoMsciEM') || 10) / 100;
  const geoData = [
    { label: 'Pays développés', value: pea.valeur * 0.85 + cto.valeur * pw, color: '#7c6dfa' },
    { label: 'Émergents', value: pea.valeur * 0.15 + cto.valeur * pe, color: '#a78bfa' },
    { label: 'Crypto (global)', value: btc.valeurActuelle + eth.valeurActuelle, color: '#fb923c' }
  ].filter(x => x.value > 0);
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  const cx = canvas.width / 2, cy = canvas.height / 2, r = 120;
  let angle = -Math.PI / 2;
  geoData.forEach(item => {
    const slice = (item.value / total) * 2 * Math.PI;
    ctx.beginPath(); ctx.fillStyle = item.color; ctx.moveTo(cx, cy); ctx.arc(cx, cy, r, angle, angle + slice); ctx.closePath(); ctx.fill();
    const mid = angle + slice / 2;
    ctx.fillStyle = '#fff'; ctx.font = 'bold 12px Outfit'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
    ctx.fillText(((item.value / total) * 100).toFixed(1) + '%', cx + Math.cos(mid) * r * 0.65, cy + Math.sin(mid) * r * 0.65);
    angle += slice;
  });
  $('geoLegend').innerHTML = geoData.map(item => `<div class="legend-item"><div class="legend-color" style="background:${item.color}"></div><span class="legend-label">${item.label}</span><span class="legend-value">${formatEuro(item.value)} (${((item.value / total) * 100).toFixed(1)}%)</span></div>`).join('');
}

function calculerRetraite() {
  const ageRetraite = $val('ageRetraite'), rente = $val('renteRetraite'), taux = $val('tauxRetraitRetraite'), age = $val('age');
  const rend = $val('rendement'), invPEA = $val('investissementPEA'), invCTO = $val('investissementCTO'), fraisPct = $val('fraisCourtage') / 100;
  const data = getCapitalTotal();
  const capitalActuel = data.pea.valeur + data.cto.valeur + data.livret + data.courant + data.especes;
  const capitalNecessaire = rente * 12 / (taux / 100);
  const mths = (ageRetraite - age) * 12;
  const capitalProj = projeterCapital(capitalActuel, (invPEA + invCTO) * (1 - fraisPct), mths, rend);
  const renteObtenue = capitalProj * (taux / 100) / 12;
  $('resultatsRetraite').innerHTML = `
    <div class="stat-card"><div class="stat-label">Capital nécessaire</div><div class="stat-value">${formatEuro(capitalNecessaire)}</div></div>
    <div class="stat-card"><div class="stat-label">Capital projeté à ${ageRetraite} ans</div><div class="stat-value">${formatEuro(capitalProj)}</div></div>
    <div class="stat-card"><div class="stat-label">${capitalProj >= capitalNecessaire ? 'Surplus' : 'Déficit'}</div><div class="stat-value" style="color:${capitalProj >= capitalNecessaire ? 'var(--green)' : 'var(--red)'}">${formatEuro(Math.abs(capitalProj - capitalNecessaire))}</div></div>
    <div class="stat-card"><div class="stat-label">Rente mensuelle possible</div><div class="stat-value">${formatEuro(renteObtenue)}/mois</div></div>`;
}

function calculerCarriere() {
  const alt = $val('salaireAlternance'), mc = $val('salaireMC'), arm = $val('salaireArmee'), av = $val('avantagesArmee');
  $('salaireCivil').textContent = formatEuroRound(mc) + '/mois';
  $('salaireArmeeTotal').textContent = formatEuroRound(arm + av) + '/mois';
  $('totalCivil').textContent = formatEuroRound(mc * 12 * 40);
  $('totalArmee').textContent = formatEuroRound((arm + av) * 12 * 25);

  const canvas = $('carriereChart');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  const pad = 60, gw = canvas.width - pad * 2, gh = canvas.height - pad * 2, age = $val('age');
  const ptsCivil = [], ptsArmee = [];
  for (let a = age; a <= 65; a += 5) {
    ptsCivil.push({ age: a, sal: a < 20 ? alt : mc + (a - 20) * 50 });
    ptsArmee.push({ age: a, sal: a <= 42 ? (arm + av + (a - age) * 30) : 0 });
  }
  const maxS = Math.max(...ptsCivil.map(p => p.sal), ...ptsArmee.map(p => p.sal), 1);
  ctx.strokeStyle = 'rgba(255,255,255,0.06)'; ctx.lineWidth = 1;
  for (let i = 0; i <= 5; i++) { const y = pad + (gh / 5) * i; ctx.beginPath(); ctx.moveTo(pad, y); ctx.lineTo(canvas.width - pad, y); ctx.stroke(); }
  [{ pts: ptsCivil, color: '#7c6dfa' }, { pts: ptsArmee, color: '#00D27F' }].forEach(line => {
    ctx.beginPath(); ctx.strokeStyle = line.color; ctx.lineWidth = 3; ctx.lineJoin = 'round';
    line.pts.forEach((p, i) => { const x = pad + (gw / (line.pts.length - 1)) * i, y = pad + gh - (p.sal / maxS) * gh; if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y); });
    ctx.stroke();
    line.pts.forEach((p, i) => { const x = pad + (gw / (line.pts.length - 1)) * i, y = pad + gh - (p.sal / maxS) * gh; ctx.beginPath(); ctx.fillStyle = line.color; ctx.arc(x, y, 4, 0, 2 * Math.PI); ctx.fill(); });
  });
  ptsCivil.forEach((p, i) => { ctx.fillStyle = 'rgba(200,196,255,0.5)'; ctx.font = '10px Outfit'; ctx.textAlign = 'center'; ctx.textBaseline = 'top'; ctx.fillText(p.age + ' ans', pad + (gw / (ptsCivil.length - 1)) * i, canvas.height - pad + 10); });
  ctx.fillStyle = '#7c6dfa'; ctx.font = 'bold 12px Outfit'; ctx.textAlign = 'left'; ctx.fillText('Civil', pad + 10, pad + 10);
  ctx.fillStyle = '#00D27F'; ctx.fillText('Armée', pad + 10, pad + 28);
}

function afficherVueSimple() {
  const data = getCapitalTotal();
  $('capitalSimple').textContent = formatEuro(data.total);
  const inv = $val('investissementPEA') + $val('investissementCTO');
  $('objectifMois').textContent = formatEuroRound(inv);
  $('prochaineAction').textContent = 'Investis ' + formatEuroRound(inv);
  const now = new Date(), jours = new Date(now.getFullYear(), now.getMonth() + 1, 0).getDate() - now.getDate();
  $('tempsRestant').textContent = jours <= 0 ? "Aujourd'hui" : 'Dans ' + jours + ' jour' + (jours > 1 ? 's' : '');

  // Graphique motivation
  const canvas = $('motivationChart');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  const allHist = [...state.historiquePEA, ...state.historiqueCTO, ...state.historiqueBTC, ...state.historiqueETH].sort((a, b) => new Date(a.date) - new Date(b.date));
  if (allHist.length === 0) { ctx.fillStyle = 'var(--text-muted)'; ctx.font = '14px DM Sans'; ctx.textAlign = 'center'; ctx.fillText('Ajoute des investissements !', canvas.width / 2, canvas.height / 2); return; }
  const grouped = {};
  allHist.forEach(h => { const key = h.date.substring(0, 7); grouped[key] = (grouped[key] || 0) + h.montant; });
  const keys = Object.keys(grouped).sort().slice(-12), vals = keys.map(k => grouped[k]), maxVal = Math.max(...vals, 1);
  const pad = 50, gw = canvas.width - pad * 2, gh = canvas.height - pad * 2, barW = Math.min(gw / keys.length - 6, 35);
  keys.forEach((k, i) => {
    const x = pad + (gw / keys.length) * i + (gw / keys.length - barW) / 2;
    const h = (vals[i] / maxVal) * gh, y = pad + gh - h;
    ctx.fillStyle = '#7c6dfa';
    ctx.beginPath(); if (ctx.roundRect) ctx.roundRect(x, y, barW, h, 4); else ctx.rect(x, y, barW, h); ctx.fill();
    ctx.fillStyle = 'rgba(200,196,255,0.55)'; ctx.font = '10px Outfit'; ctx.textAlign = 'center';
    ctx.fillText(k.substring(5), x + barW / 2, canvas.height - pad + 14);
    ctx.fillStyle = 'rgba(240,240,255,0.9)'; ctx.font = 'bold 10px Outfit';
    ctx.fillText(formatEuroRound(vals[i]), x + barW / 2, y - 8);
  });
}

function ajouterRevenu() {
  const mois = $('revenuMois').value, annee = $val('revenuAnnee'), montant = $val('revenuMontant'), source = $('revenuSource').value || 'Revenu';
  if (!annee || montant <= 0) { alert('❌ Remplis l\'année et le montant !'); return; }
  const key = annee + '-' + mois;
  const existing = state.historiqueRevenus.findIndex(x => x.key === key);
  if (existing >= 0) { if (!confirm('Remplacer le revenu existant pour ce mois ?')) return; state.historiqueRevenus.splice(existing, 1); }
  state.historiqueRevenus.push({ key, mois, annee, montant, source, id: Date.now() });
  state.historiqueRevenus.sort((a, b) => a.key.localeCompare(b.key));
  $('revenuMontant').value = ''; $('revenuSource').value = '';
  afficherRevenus(); sauvegarder();
}

function supprimerRevenu(id) {
  if (!confirm('Supprimer ?')) return;
  state.historiqueRevenus = state.historiqueRevenus.filter(x => x.id !== id);
  afficherRevenus(); sauvegarder();
}

function getInvestiDuMois(key) {
  return [...state.historiquePEA, ...state.historiqueCTO, ...state.historiqueBTC, ...state.historiqueETH]
    .filter(x => x.date && x.date.substring(0, 7) === key)
    .reduce((s, x) => s + x.montant, 0);
}

function afficherRevenus() {
  const grid = $('revenuStatsGrid'), tbody = $('historiqueRevenus');
  if (!grid || !tbody) return;
  const nomsMois = ['', 'Jan', 'Fév', 'Mar', 'Avr', 'Mai', 'Jun', 'Jul', 'Aoû', 'Sep', 'Oct', 'Nov', 'Déc'];

  if (state.historiqueRevenus.length === 0) {
    grid.innerHTML = '<div style="text-align:center;color:var(--text-muted);padding:32px;grid-column:1/-1;background:var(--bg-glass);border-radius:var(--radius-lg);border:1px solid var(--border-subtle)">💡 Aucun revenu — ajoute ton premier salaire !</div>';
    tbody.innerHTML = '<tr><td colspan="7" style="text-align:center;color:var(--text-muted);padding:24px">Aucun revenu</td></tr>';
    return;
  }

  const totalGagne = state.historiqueRevenus.reduce((s, r) => s + r.montant, 0);
  const count = state.historiqueRevenus.length;
  const moy = totalGagne / count;
  let totalInv = 0;
  state.historiqueRevenus.forEach(r => { totalInv += getInvestiDuMois(r.key); });
  const tauxEp = totalGagne > 0 ? (totalInv / totalGagne) * 100 : 0;

  grid.innerHTML = `
    <div class="stat-card revenus"><div class="stat-label">💰 Total gagné</div><div class="stat-value" style="color:var(--green)">${formatEuro(totalGagne)}</div></div>
    <div class="stat-card"><div class="stat-label">📊 Moyenne mensuelle</div><div class="stat-value">${formatEuro(moy)}</div></div>
    <div class="stat-card pea"><div class="stat-label">📈 Total investi</div><div class="stat-value" style="color:var(--accent-soft)">${formatEuro(totalInv)}</div></div>
    <div class="stat-card"><div class="stat-label">💼 Taux d'épargne</div><div class="stat-value" style="color:${tauxEp >= 20 ? 'var(--green)' : 'var(--orange)'}">${tauxEp.toFixed(1)}%</div></div>
    <div class="stat-card"><div class="stat-label">📅 Mois enregistrés</div><div class="stat-value">${count}</div></div>
    <div class="stat-card"><div class="stat-label">🍽️ Dépenses estimées</div><div class="stat-value" style="color:var(--red)">${formatEuro(totalGagne - totalInv)}</div></div>`;

  tbody.innerHTML = [...state.historiqueRevenus].reverse().map(r => {
    const inv = getInvestiDuMois(r.key), dep = r.montant - inv, taux = r.montant > 0 ? (inv / r.montant) * 100 : 0;
    return `<tr><td><strong>${nomsMois[Number(r.mois)]} ${r.annee}</strong></td><td><span style="background:rgba(107,87,255,0.1);border:1px solid rgba(107,87,255,0.2);padding:2px 10px;border-radius:99px;font-size:11px;color:var(--accent-soft);font-weight:600">${r.source}</span></td><td class="positive"><strong>${formatEuro(r.montant)}</strong></td><td style="color:var(--accent-soft)">${inv > 0 ? formatEuro(inv) : '--'}</td><td style="color:var(--red)">${dep > 0 ? formatEuro(dep) : '--'}</td><td><span style="color:${taux >= 20 ? 'var(--green)' : 'var(--orange)'};font-weight:700">${taux > 0 ? taux.toFixed(1) + '%' : '--'}</span></td><td><button class="btn-delete" onclick="supprimerRevenu(${r.id})">🗑️</button></td></tr>`;
  }).join('');

  // Graphique revenus
  const c1 = $('revenuChart');
  if (c1 && state.historiqueRevenus.length > 0) {
    const ctx = c1.getContext('2d'); ctx.clearRect(0, 0, c1.width, c1.height);
    const data = state.historiqueRevenus, maxV = Math.max(...data.map(d => d.montant), 1);
    const pad = 55, gw = c1.width - pad * 2, gh = c1.height - pad * 2, barW = Math.min(gw / data.length - 4, 42);
    data.forEach((d, i) => {
      const x = pad + (gw / data.length) * i + (gw / data.length - barW) / 2;
      const h = (d.montant / maxV) * gh, y = pad + gh - h;
      const grad = ctx.createLinearGradient(0, y, 0, pad + gh);
      grad.addColorStop(0, 'rgba(0,210,127,0.8)'); grad.addColorStop(1, 'rgba(0,210,127,0.15)');
      ctx.fillStyle = grad; ctx.beginPath(); if (ctx.roundRect) ctx.roundRect(x, y, barW, h, 4); else ctx.rect(x, y, barW, h); ctx.fill();
      ctx.fillStyle = 'rgba(200,196,255,0.45)'; ctx.font = '9px Outfit'; ctx.textAlign = 'center'; ctx.textBaseline = 'top';
      ctx.fillText(nomsMois[Number(d.mois)] + "'" + d.annee.toString().slice(2), x + barW / 2, c1.height - pad + 6);
      if (barW > 18) { ctx.fillStyle = 'rgba(240,240,255,0.85)'; ctx.font = 'bold 9px Outfit'; ctx.textBaseline = 'bottom'; ctx.fillText((d.montant / 1000).toFixed(1) + 'k', x + barW / 2, y - 5); }
    });
  }

  // Graphique revenus vs invest
  const c2 = $('revenuInvChart');
  if (c2 && state.historiqueRevenus.length > 0) {
    const ctx = c2.getContext('2d'); ctx.clearRect(0, 0, c2.width, c2.height);
    const n = state.historiqueRevenus.length;
    const revV = state.historiqueRevenus.map(r => r.montant);
    const invV = state.historiqueRevenus.map(r => getInvestiDuMois(r.key));
    const maxV = Math.max(...revV, ...invV, 1);
    const pad = 55, gw = c2.width - pad * 2, gh = c2.height - pad * 2, barGroup = gw / n, barW = Math.min((barGroup - 8) / 2, 24);
    state.historiqueRevenus.forEach((r, i) => {
      const gx = pad + i * barGroup + barGroup / 2;
      const hR = (revV[i] / maxV) * gh, yR = pad + gh - hR;
      const gR = ctx.createLinearGradient(0, yR, 0, pad + gh); gR.addColorStop(0, 'rgba(107,87,255,0.75)'); gR.addColorStop(1, 'rgba(107,87,255,0.2)');
      ctx.fillStyle = gR; ctx.beginPath(); if (ctx.roundRect) ctx.roundRect(gx - barW - 2, yR, barW, hR, 3); else ctx.rect(gx - barW - 2, yR, barW, hR); ctx.fill();
      if (invV[i] > 0) {
        const hI = (invV[i] / maxV) * gh, yI = pad + gh - hI;
        const gI = ctx.createLinearGradient(0, yI, 0, pad + gh); gI.addColorStop(0, 'rgba(0,210,127,0.75)'); gI.addColorStop(1, 'rgba(0,210,127,0.15)');
        ctx.fillStyle = gI; ctx.beginPath(); if (ctx.roundRect) ctx.roundRect(gx + 2, yI, barW, hI, 3); else ctx.rect(gx + 2, yI, barW, hI); ctx.fill();
      }
      if (n <= 16) {
        ctx.fillStyle = 'rgba(200,196,255,0.45)'; ctx.font = '9px Outfit'; ctx.textAlign = 'center'; ctx.textBaseline = 'top';
        ctx.fillText(nomsMois[Number(r.mois)] + "'" + r.annee.toString().slice(2), gx, c2.height - pad + 6);
      }
    });
  }
}
