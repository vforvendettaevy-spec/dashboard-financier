function formatEuroDecimal(val,signe=false){
  if(val==null||isNaN(val))return'0,00 €';
  const fmt=val.toLocaleString('fr-FR',{minimumFractionDigits:2,maximumFractionDigits:2})+' €';
  return signe&&val>=0?'+ '+fmt:fmt;
}
function formatEuro(val,signe=false){
  if(val==null||isNaN(val))return'0 €';
  const fmt=Math.round(val).toLocaleString('fr-FR')+' €';
  return signe&&val>=0?'+ '+fmt:fmt;
}

function calculerValeurPEA(rendementAnn){
  const today=new Date();const mRate=(rendementAnn/100)/12;
  const fraisPct=(Number(document.getElementById('fraisCourtage').value)||0)/100;
  let valeur=0,totalVerse=0,totalFrais=0;
  historiquePEA.forEach(item=>{
    const frais=item.montant*fraisPct;totalFrais+=frais;totalVerse+=item.montant;
    if(item.partsWPEA!=null&&item.partsPAEEM!=null&&coursWPEA>0&&coursPAEEM>0){
      valeur+=item.partsWPEA*coursWPEA+item.partsPAEEM*coursPAEEM;
    }else{
      const dep=new Date(item.date);let m=(today.getFullYear()-dep.getFullYear())*12+(today.getMonth()-dep.getMonth());
      if(m<0)m=0;valeur+=(item.montant-frais)*Math.pow(1+mRate,m);
    }
  });
  return{valeur,totalVerse,plusValue:valeur-(totalVerse-totalFrais),totalFrais};
}

function calculerValeurCTO(rendementAnn){
  const today=new Date();const mRate=(rendementAnn/100)/12;
  const fraisPct=(Number(document.getElementById('fraisCourtage').value)||0)/100;
  let valeur=0,totalVerse=0,totalFrais=0;
  historiqueCTO.forEach(item=>{
    const frais=item.montant*fraisPct;totalFrais+=frais;totalVerse+=item.montant;
    if(item.partsWPEA!=null&&item.partsPAEEM!=null&&coursEUNL>0&&coursIS3N>0){
      valeur+=item.partsWPEA*coursEUNL+item.partsPAEEM*coursIS3N;
    }else{
      const dep=new Date(item.date);let m=(today.getFullYear()-dep.getFullYear())*12+(today.getMonth()-dep.getMonth());
      if(m<0)m=0;valeur+=(item.montant-frais)*Math.pow(1+mRate,m);
    }
  });
  return{valeur,totalVerse,plusValue:valeur-(totalVerse-totalFrais),totalFrais};
}

function calculerCrypto(historique,coursActuel){
  const fraisPct=(Number(document.getElementById('fraisCourtage').value)||0)/100;
  let totalInvesti=0,quantiteTotale=0,totalFrais=0;
  historique.forEach(item=>{
    const frais=item.montant*fraisPct;const net=item.montant-frais;
    quantiteTotale+=net/item.prixAchat;totalInvesti+=item.montant;totalFrais+=frais;
  });
  const valeurActuelle=quantiteTotale*coursActuel;
  return{totalInvesti,quantiteTotale,valeurActuelle,plusValue:valeurActuelle-(totalInvesti-totalFrais),totalFrais};
}

function projectionCapital(capitalActuel,invMensuel,rendAnn,mois){
  const mRate=(rendAnn/100)/12;
  if(mRate===0) return capitalActuel+invMensuel*mois;
  return capitalActuel*Math.pow(1+mRate,mois)+invMensuel*((Math.pow(1+mRate,mois)-1)/mRate);
}

function dessinerPie(valPEA,valCTO,valCrypto,livret,courant,soldeEspeces){
  const canvas=document.getElementById('pieChart');
  if(!canvas)return;
  const ctx=canvas.getContext('2d');
  const total=valPEA+valCTO+valCrypto+livret+courant+soldeEspeces;
  const legend=document.getElementById('pieLegend');
  if(total===0){
    ctx.clearRect(0,0,canvas.width,canvas.height);
    ctx.fillStyle='rgba(200,196,255,0.35)';ctx.font='14px "DM Sans",sans-serif';
    ctx.textAlign='center';ctx.fillText('Commence à investir !',canvas.width/2,canvas.height/2);
    legend.innerHTML='';return;
  }
  const items=[
    {label:'📈 PEA',value:valPEA,color:'#7c6dfa'},
    {label:'💵 PEA Espèces',value:soldeEspeces,color:'#a78bfa'},
    {label:'📊 CTO',value:valCTO,color:'#6366f1'},
    {label:'₿ Crypto',value:valCrypto,color:'#fb923c'},
    {label:'💚 Livret A',value:livret,color:'#00d4aa'},
    {label:'💳 Compte',value:courant,color:'#60a5fa'}
  ].filter(x=>x.value>0);
  ctx.clearRect(0,0,canvas.width,canvas.height);
  const cx=canvas.width/2,cy=canvas.height/2,r=108;
  let angle=-Math.PI/2;
  items.forEach(item=>{
    const slice=(item.value/total)*2*Math.PI;
    ctx.beginPath();ctx.fillStyle=item.color;ctx.moveTo(cx,cy);
    ctx.arc(cx,cy,r,angle,angle+slice);ctx.closePath();ctx.fill();
    const mid=angle+slice/2;
    const pct=((item.value/total)*100).toFixed(1);
    if(slice>0.2){
      ctx.fillStyle='#fff';ctx.font='bold 11px "Outfit",sans-serif';
      ctx.textAlign='center';ctx.textBaseline='middle';
      ctx.fillText(pct+'%',cx+Math.cos(mid)*r*0.65,cy+Math.sin(mid)*r*0.65);
    }
    angle+=slice;
  });
  legend.innerHTML=items.map(item=>{
    const pct=((item.value/total)*100).toFixed(1);
    return`<div class="legend-item"><div class="legend-color" style="background:${item.color}"></div><span class="legend-label">${item.label}</span><span class="legend-value">${formatEuroDecimal(item.value)} (${pct}%)</span></div>`;
  }).join('');
}

function afficherProjection(capitalActuel,rend){
  const ages=[20,25,30,35,40,45,50,55,60,65];
  const {age,invPEA,invCTO,fraisPct}=getParams();
  const invMensuel=(invPEA+invCTO)*(1-fraisPct);
  const grid=document.getElementById('projectionGrid');
  const tbody=document.getElementById('projectionBody');
  if(!grid||!tbody)return;
  grid.innerHTML='';tbody.innerHTML='';
  ages.filter(a=>a>age).forEach(a=>{
    const mths=(a-age)*12;
    const cap=projectionCapital(capitalActuel,invMensuel,rend,mths);
    const verse=capitalActuel+(invPEA+invCTO)*mths;
    const pv=cap-verse+(invPEA+invCTO)*mths*fraisPct;
    const isTarget=a===settings.objectifAge;
    grid.innerHTML+=`<div class="projection-card" ${isTarget?'style="border-color:rgba(0,210,127,0.4);background:rgba(0,210,127,0.06)"':''}>
      <div class="projection-age">${a} ans</div>
      <div class="projection-amount" style="font-size:${isTarget?'14':'12'}px;${isTarget?'color:#00D27F;font-weight:700':''}">
        ${formatEuroDecimal(cap)}
      </div>
    </div>`;
    tbody.innerHTML+=`<tr>
      <td><strong>${a} ans</strong>${isTarget?' <span style="font-size:10px;background:rgba(0,210,127,0.1);color:var(--green);padding:2px 6px;border-radius:4px">🎯 Objectif</span>':''}</td>
      <td>${a-age} an${a-age>1?'s':''}</td>
      <td>${formatEuroDecimal(verse)}</td>
      <td class="positive"><strong>${formatEuroDecimal(cap)}</strong></td>
      <td class="${pv>=0?'positive':'negative'}">${formatEuroDecimal(pv,true)}</td>
    </tr>`;
  });
}

function calculerScenarios(capitalActuel,rendActuel){
  const {age,invPEA,invCTO,fraisPct}=getParams();
  const invMensuel=(invPEA+invCTO)*(1-fraisPct);
  const yrs=settings.objectifAge-age;
  if(yrs<=0)return;
  const mths=yrs*12;
  [{id:'Pessimiste',r:5},{id:'Neutre',r:6.5},{id:'Optimiste',r:8},{id:'UltraOptimiste',r:9.5}].forEach(s=>{
    const cap=projectionCapital(capitalActuel,invMensuel,s.r,mths);
    const gains=cap-capitalActuel-(invPEA+invCTO)*mths+(invPEA+invCTO)*mths*fraisPct;
    document.getElementById('scenario'+s.id).textContent=formatEuroDecimal(cap);
    document.getElementById('detail'+s.id).textContent='+'+formatEuroDecimal(gains)+' gains';
  });
}

function calculerDCA(){

  historiquePEA.forEach(it=>{if(it.partsWPEA>0){tPW+=it.partsWPEA;tiW+=it.partsWPEA*it.prixAchatWPEA}if(it.partsPAEEM>0){tPP+=it.partsPAEEM;tiP+=it.partsPAEEM*it.prixAchatPAEEM}});
  let tPE=0,tiE=0,tPI=0,tiI=0;
  historiqueCTO.forEach(it=>{if(it.partsWPEA>0){tPE+=it.partsWPEA;tiE+=it.partsWPEA*it.prixAchatWPEA}if(it.partsPAEEM>0){tPI+=it.partsPAEEM;tiI+=it.partsPAEEM*it.prixAchatPAEEM}});

  let tiB=0,tqB=0;historiqueBTC.forEach(it=>{const n=it.montant*(1-fp);tqB+=n/it.prixAchat;tiB+=n});
  let tiEH=0,tqEH=0;historiqueETH.forEach(it=>{const n=it.montant*(1-fp);tqEH+=n/it.prixAchat;tiEH+=n});
  document.getElementById('dcaWPEA').textContent=tPW>0?formatEuroDecimal(tiW/tPW):'-- €';
  document.getElementById('dcaPAEEM').textContent=tPP>0?formatEuroDecimal(tiP/tPP):'-- €';
  document.getElementById('dcaWPEACTO').textContent=tPE>0?formatEuroDecimal(tiE/tPE):'-- €';
  document.getElementById('dcaPAEEMCTO').textContent=tPI>0?formatEuroDecimal(tiI/tPI):'-- €';
  document.getElementById('dcaBTC').textContent=tqB>0?formatEuroDecimal(tiB/tqB):'-- €';
  document.getElementById('dcaETH').textContent=tqEH>0?formatEuroDecimal(tiEH/tqEH):'-- €';
}

function calculerFIRE(){
  const dep=Number(document.getElementById('fireDepenses').value)||40000;
  const taux=Number(document.getElementById('fireTaux').value)||4;

  const capNec=dep/(taux/100);const manquant=capNec-capActuel;

    <div class="stat-card"><div class="stat-label">🍽️ Total dépenses estimées</div><div class="stat-value" style="color:var(--red)">${formatEuroDecimal(totalDep)}</div></div>
    <div class="stat-card"><div class="stat-label">📅 Mois enregistrés</div><div class="stat-value">${moisCount}</div></div>
    <div class="stat-card"><div class="stat-label">🔝 Meilleur mois</div><div class="stat-value positive">${formatEuroDecimal(maxR)}</div></div>
    <div class="stat-card"><div class="stat-label">📉 Plus petit mois</div><div class="stat-value">${formatEuroDecimal(minR)}</div></div>
    <div class="stat-card"><div class="stat-label">💸 Invest. moyen/mois</div><div class="stat-value" style="color:var(--accent-soft)">${formatEuroDecimal(invMoyM)}</div></div>
    <div class="stat-card"><div class="stat-label">📈 Évolution salaire</div><div class="stat-value" style="color:${evol>=0?'var(--green)':'var(--red)'}">${evol>=0?'+':''}${evol.toFixed(1)}%</div></div>

    const inv=getInvestiDuMois(r.key);const dep=r.montant-inv;const tx=r.montant>0?(inv/r.montant)*100:0;
    return`<tr>
      <td><strong>${MOIS_NOMS[Number(r.mois)]} ${r.annee}</strong></td>

      <td class="positive"><strong>${formatEuroDecimal(r.montant)}</strong></td>

      <td style="color:var(--red)">${dep>0?formatEuroDecimal(dep):'<span style="color:var(--text-ultra-muted)">--</span>'}</td>
      <td><span style="background:${tx>=20?'rgba(0,210,127,0.1)':tx>0?'rgba(255,140,66,0.1)':'rgba(255,255,255,0.03)'};border:1px solid ${tx>=20?'rgba(0,210,127,0.2)':tx>0?'rgba(255,140,66,0.2)':'rgba(255,255,255,0.05)'};padding:3px 10px;border-radius:99px;font-size:12px;color:${tx>=20?'var(--green)':tx>0?'var(--orange)':'var(--text-muted)'};font-weight:700">${tx>0?tx.toFixed(1)+'%':'--'}</span></td>
      <td><button class="btn-delete" onclick="supprimerRevenu(${r.id})">🗑️</button></td>
    </tr>`;
  }).join('');
  dessinerRevenuChart();dessinerRevenuInvChart();
}
